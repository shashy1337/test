def score(y_true, y_pred):
    r2 = r2_score(y_true, y_pred)
    mae = mean_absolute_error(y_true, y_pred)
    print("R^2: " + str(r2))
    print("-" * 100)
    print("MAE: " + str(mae))
    print("-" * 100)
